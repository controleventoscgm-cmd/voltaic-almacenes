SECCION 1: CREAR EL PRIMER ADMINISTRADOR EN FIREBASE CONSOLE
Paso a paso para que solo tu tengas acceso al panel:
Paso 1: Entra a Firebase Console
URL: https://console.firebase.google.com
Inicia sesion con tu cuenta de Google (la que usaras como admin)
Paso 2: Crea un proyecto nuevo
Click en "Agregar proyecto"
Nombre del proyecto: [tu-app-negocios-barrio]
Desactiva Google Analytics (no lo necesitas ahora)
Click en "Crear proyecto"
Paso 3: Activa Firestore Database
En el menu lateral, click en "Firestore Database"
Click en "Crear base de datos"
Selecciona "modo de prueba" por ahora (luego cambiamos a produccion)
Elige la region: us-central1 (o la mas cercana)
Paso 4: Activa Authentication
En el menu lateral, click en "Authentication"
Click en "Comenzar"
Activa "Google" como proveedor de inicio de sesion
Guarda la configuracion
Paso 5: Crea tu usuario admin manualmente
Ve a Firestore Database
Crea una coleccion llamada "administradores"
Crea un documento con el ID igual a tu UID de Google
Para obtener tu UID:
Ve a Authentication > Users
Inicia sesion una vez con tu Google en la app
Apareceras en la lista de usuarios
Copia tu UID (es un string largo como: AbC123XyZ789)
Paso 6: Crea el documento admin
Coleccion: administradores
Documento ID: [tu UID copiado]
Campos del documento:
rol: string = "super_admin"
email: string = "tuemail@gmail.com"
nombre: string = "Tu Nombre"
creado: timestamp = [fecha de hoy]
Paso 7: Configura las Firestore Rules
Ve a Firestore Database > Rules
Pega las reglas que te di en la respuesta anterior
Click en "Publicar"
Paso 8: Obtén las credenciales de Firebase para tu app
Ve a Configuracion del proyecto (icono de engranaje)
Click en "Configuracion"
En "Tus apps", click en "</>" para crear una app web
Registra el app con un nickname
Copia el objeto firebaseConfig que te muestra
Pega ese objeto en tu archivo firebase-config.js
Paso 9: Activa Firebase Hosting (para el panel admin)
Ve a Hosting en el menu lateral
Click en "Comenzar"
Sigue las instrucciones para instalar Firebase CLI
Despliega tu carpeta /admin-panel
Paso 10: Prueba el login
Abre tu panel en la URL que te da Firebase Hosting
Intenta iniciar sesion con tu Google
Si configuraste bien, entras al dashboard
Si intentas con otra cuenta Google, te rechaza

// Verificar admin al cargar
auth.onAuthStateChanged(async (user) => {
    if (!user) {
        window.location.href = 'index.html';
        return;
    }
    
    // Verificar que sea admin
    const adminDoc = await db.collection('administradores').doc(user.uid).get();
    if (!adminDoc.exists || adminDoc.data().rol !== 'super_admin') {
        await auth.signOut();
        window.location.href = 'index.html';
        return;
    }
    
    adminUid = user.uid;
    document.getElementById('admin-email').textContent = user.email;
    
    // Cargar datos
    await cargarResumen();
    await cargarLicencias();
});

function cerrarSesion() {
    auth.signOut();
}

async function cargarResumen() {
    const hoy = new Date();
    hoy.setHours(0, 0, 0, 0);
    
    const sieteDias = new Date(hoy);
    sieteDias.setDate(sieteDias.getDate() + 7);
    
    try {
        const snapshot = await db.collection('licencias').get();
        
        let activas = 0;
        let porVencer = 0;
        let vencidas = 0;
        let ingresoMensual = 0;
        
        snapshot.forEach(doc => {
            const lic = doc.data();
            const fechaTermino = lic.fecha_termino ? new Date(lic.fecha_termino) : null;
            
            // Actualizar estado si esta vencida
            if (fechaTermino && fechaTermino < hoy && lic.estado !== 'vencida') {
                db.collection('licencias').doc(doc.id).update({ estado: 'vencida' });
                lic.estado = 'vencida';
            }
            
            if (lic.estado === 'activa') {
                activas++;
                
                // Calcular ingreso mensual equivalente
                switch(lic.tipo_plan) {
                    case 'mensual': ingresoMensual += 8900; break;
                    case 'trimestral': ingresoMensual += 8300; break; // 24900/3
                    case 'semestral': ingresoMensual += 7900; break; // 47400/6
                    case 'anual': ingresoMensual += 7417; break; // 89000/12
                }
                
                if (fechaTermino && fechaTermino <= sieteDias && fechaTermino >= hoy) {
                    porVencer++;
                }
            }
            
            if (lic.estado === 'vencida') {
                vencidas++;
            }
        });
        
        document.getElementById('total-activas').textContent = activas;
        document.getElementById('total-por-vencer').textContent = porVencer;
        document.getElementById('total-vencidas').textContent = vencidas;
        document.getElementById('ingreso-mensual').textContent = '$' + ingresoMensual.toLocaleString('es-CL');
        
    } catch (error) {
        console.error('Error cargando resumen:', error);
        mostrarNotificacion('Error al cargar datos', 'error');
    }
}

async function cargarLicencias() {
    try {
        const snapshot = await db.collection('licencias')
            .orderBy('fecha_creacion', 'desc')
            .get();
        
        licenciasData = [];
        const tbody = document.getElementById('tbody-licencias');
        tbody.innerHTML = '';
        
        snapshot.forEach(doc => {
            const lic = doc.data();
            lic.id = doc.id;
            licenciasData.push(lic);
            
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td>${lic.id}</td>
                <td>${lic.negocio || '-'}</td>
                <td>${lic.tipo_plan || '-'}</td>
                <td>${formatearFecha(lic.fecha_inicio)}</td>
                <td>${formatearFecha(lic.fecha_termino)}</td>
                <td><span class="estado ${lic.estado}">${lic.estado}</span></td>
                <td>
                    <button class="btn-small btn-edit" onclick="editarLicencia('${lic.id}')">Editar</button>
                    <button class="btn-small btn-renew" onclick="renovarLicencia('${lic.id}')">Renovar</button>
                    <button class="btn-small btn-delete" onclick="eliminarLicencia('${lic.id}')">Eliminar</button>
                </td>
            `;
            tbody.appendChild(tr);
        });
        
    } catch (error) {
        console.error('Error cargando licencias:', error);
        mostrarNotificacion('Error al cargar licencias', 'error');
    }
}

async function crearNuevaLicencia() {
    try {
        // Generar ID secuencial
        const contadorRef = db.collection('config').doc('contador');
        const contadorDoc = await contadorRef.get();
        
        let nuevoNumero = 1;
        if (contadorDoc.exists) {
            nuevoNumero = contadorDoc.data().ultimo_numero + 1;
            await contadorRef.update({ 
                ultimo_numero: nuevoNumero,
                ultima_actualizacion: firebase.firestore.FieldValue.serverTimestamp()
            });
        } else {
            await contadorRef.set({ 
                ultimo_numero: 1,
                ultima_actualizacion: firebase.firestore.FieldValue.serverTimestamp()
            });
        }
        
        const licenciaId = `LIC-2026-${String(nuevoNumero).padStart(6, '0')}`;
        const codigo = generarCodigoAleatorio(10);

        const fechaInicio = new Date(document.getElementById('fecha-inicio').value);
        const fechaTermino = new Date(fechaInicio);
        
        switch(tipoPlan) {
            case 'prueba': fechaTermino.setDate(fechaTermino.getDate() + 30); break;
            case 'mensual': fechaTermino.setDate(fechaTermino.getDate() + 30); break;
            case 'trimestral': fechaTermino.setDate(fechaTermino.getDate() + 90); break;
            case 'semestral': fechaTermino.setDate(fechaTermino.getDate() + 180); break;
            case 'anual': fechaTermino.setDate(fechaTermino.getDate() + 365); break;
        }
        
        const nuevaLicencia = {
            codigo: codigo,
            negocio: document.getElementById('negocio').value.trim(),

            fecha_pago: document.getElementById('monto-pagado').value > 0

                : null,

            metodo_pago: document.getElementById('metodo-pago').value,
            estado: 'activa',
            notas: document.getElementById('notas').value.trim() || null,
            creado_por: adminUid,
            fecha_creacion: firebase.firestore.FieldValue.serverTimestamp(),
            ultima_modificacion: firebase.firestore.FieldValue.serverTimestamp()
        };

        await cargarLicencias();

        enviarCodigoWhatsApp(nuevaLicencia.telefono, codigo, nuevaLicencia.negocio);
        
    } catch (error) {
        console.error('Error creando licencia:', error);
        mostrarNotificacion('Error al crear licencia', 'error');
    }
}

function generarCodigoAleatorio(longitud) {
    const caracteres = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    let resultado = '';
    for (let i = 0; i < longitud; i++) {
        resultado += caracteres.charAt(Math.floor(Math.random() * caracteres.length));
    }
    return resultado;
}

function filtrarLicencias() {
    const filtroNegocio = document.getElementById('filtro-negocio').value.toLowerCase();
    const filtroEstado = document.getElementById('filtro-estado').value;
    
    const tbody = document.getElementById('tbody-licencias');
    tbody.innerHTML = '';
    
    const filtradas = licenciasData.filter(lic => {
        const coincideNegocio = !filtroNegocio || (lic.negocio && lic.negocio.toLowerCase().includes(filtroNegocio));
        const coincideEstado = filtroEstado === 'todas' || lic.estado === filtroEstado;
        return coincideNegocio && coincideEstado;
    });
    
    filtradas.forEach(lic => {
        const tr = document.createElement('tr');
        tr.innerHTML = `
            <td>${lic.id}</td>
            <td>${lic.negocio || '-'}</td>
            <td>${lic.tipo_plan || '-'}</td>
            <td>${formatearFecha(lic.fecha_inicio)}</td>
            <td>${formatearFecha(lic.fecha_termino)}</td>
            <td><span class="estado ${lic.estado}">${lic.estado}</span></td>
            <td>
                <button class="btn-small btn-edit" onclick="editarLicencia('${lic.id}')">Editar</button>
                <button class="btn-small btn-renew" onclick="renovarLicencia('${lic.id}')">Renovar</button>
                <button class="btn-small btn-delete" onclick="eliminarLicencia('${lic.id}')">Eliminar</button>
            </td>
        `;
        tbody.appendChild(tr);
    });
}

function formatearFecha(fechaString) {
    if (!fechaString) return '-';
    const fecha = new Date(fechaString);
    return fecha.toLocaleDateString('es-CL', {
        day: '2-digit',
        month: '2-digit',
        year: 'numeric'
    });
}

    notif.className = 'notification ' + tipo;
    notif.style.display = 'block';
    
    setTimeout(() => {

    }, 4000);
}
